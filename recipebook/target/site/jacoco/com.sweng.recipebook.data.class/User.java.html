import com.sweng.recipebook.data.class_interface.IUser;

public class User implements IUser {
    private String FirstName;
    private String LastName;
    private int UserId;
    private String UserName;
    
    public User() {
        FirstName = "";
        LastName = "";
        UserId = 0;
        UserName = "";
    }

    public User(String firstName, String lastName, int userId, String userName) {
        FirstName = firstName;
        LastName = lastName;
        UserId = userId;
        UserName = userName;
    }

    public String GetFirstName() {
        return FirstName;
    }

    public String GetLastName() {
        return LastName;
    }

    public int GetUserId() {
        return UserId;
    }

    public String GetUserName() {
        return UserName;
    }
}
